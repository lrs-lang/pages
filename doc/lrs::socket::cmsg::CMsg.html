Enum lrs::socket::cmsg::CMsg

A generic control message.

Syntax

enum CMsg<'a> {
    Credentials(&'a Credentials),
    Fds(&'a [i32]),
    Ipv4Options(&'a [u8]),
    Timestamp(&'a Timestamp),
    Unknown(&'a [u8]),
}
