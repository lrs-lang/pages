Trait lrs::ops::Eq

trait Eq<Rhs = Self>
    where Rhs: ?Sized,
{
    /* Required methods */
    fn eq(&self, other: &Rhs) -> bool

    /* Provided methods */
    fn ne(&self, other: &Rhs) -> bool
}
