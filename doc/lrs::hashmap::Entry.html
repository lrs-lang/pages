Enum lrs::hashmap::Entry

A bucket in a hash map.

Syntax

enum Entry<'a, Key, Value, Bucket>
    where Key: Eq<Key> + Hash,
          Bucket: Bucket<Key, Value> + 'a,
{
    Occupied(OccupiedEntry<'a, Key, Value, Bucket>),
    Vacant(VacantEntry<'a, Key, Value, Bucket>),
}
